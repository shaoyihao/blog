炸弹

· 2022-09-13 · # 前缀和

原题链接：激光炸弹

#include <iostream>
using namespace std;

const int N = 5010;
int s[N][N];

int main()
{
    int cnt, R;
    cin >> cnt >> R;
    R = min(5001, R); //缩小R的范围

    while (cnt --)
    {
        int x, y, w;
        cin >> x >> y >> w; 
        s[x + 1][y + 1] += w; //本题中x,y从0开始，因此存储时下标+1
        //因为不同目标可能在同一位置，所以此处为 '+='
    }

    int n = 5001, m = 5001;
    for (int i = 1; i <= n; i ++)
        for (int j = 1; j <= m; j ++)
            s[i][j] += s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1]; // 预处理前缀和

    int res = 0;
    for (int i = R; i <= n; i ++) // 暴力枚举所有边长是R的矩形，枚举(i, j)为右下角
        for (int j = R; j <= m; j ++)
            res = max(res, s[i][j] - s[i - R][j] - s[i][j - R] + s[i - R][j - R]);

    cout << res;
    return 0;
}
