棋子移动

· 2022-09-04 · # 递归

原题链接：棋子移动

在处理长度为 nnn 的串时，其中包含了对 长度为 n−1n-1n−1 的串的处理，从而可以递归求解。

n=4n=4n=4 时直接打表即可。

#include <iostream>
#include <string>
using namespace std;

int n;

void print(string str, int t)
{
    cout << str << endl;

    if (t == 4)
    {
        string s;
        for (int i = 0; i < n - 4; i ++) s += "o*";
        cout << "ooo--***o*" + s << endl;
        cout << "ooo*o**--*" + s << endl;
        cout << "o--*o**oo*" + s << endl;
        cout << "o*o*o*--o*" + s << endl;
        cout << "--o*o*o*o*" + s << endl;
        return;
    }

    swap(str[t - 1], str[2 * t]);
    swap(str[t], str[2 * t + 1]);
    cout << str << endl;

    swap(str[t - 1], str[2 * (t - 1)]);
    swap(str[t], str[2 * (t - 1) + 1]);
    print(str, t - 1);
}

int main()
{
    cin >> n;
    string str;
    for (int i = 1; i <= n; i ++) str += "o";
    for (int i = 1; i <= n; i ++) str += "*";
    str += "--";

    print(str, n);

    return 0;
}
