扫雷

· 2022-09-09 · # 模拟

原题链接：扫雷

给出 nnn 行 mmm 列格子中的地雷分布，计算每个非地雷格中的数字。

#include <iostream>
using namespace std;

int dx[8] = {-1, -1, 0, 1, 1, 1, 0, -1}, dy[8] = {0, 1, 1, 1, 0, -1, -1, -1};

const int N = 210;
char g[N][N];
int q[N][N];

int main()
{
    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; i ++) 
        for (int j = 1; j <= m; j ++)
            cin >> g[i][j];
    
    for (int i = 1; i <= n; i ++)
        for (int j = 1; j <= m; j ++)
            if (g[i][j] != '*')
            {
                int cnt = 0;
                for (int k = 0; k < 8; k ++)
                    if (g[i + dx[k]][j + dy[k]] == '*') cnt ++;
                q[i][j] = cnt;
            }

    for (int i = 1; i <= n; i ++)
    {
        for (int j = 1; j <= m; j ++)
            if (g[i][j] == '*') cout << '*';
            else cout << q[i][j];
        puts("");
    }
    
    return 0;
}
