分形图

· 2022-09-04 · # 模拟 # 分治

原题链接：南蛮图腾

题目大意：已知 n=1n=1n=1时：

 /\
/__\

n=2n=2n=2时：

   /\
  /__\
 /\  /\
/__\/__\

n=3n=3n=3时：

       /\
      /__\
     /\  /\
    /__\/__\
   /\      /\
  /__\    /__\
 /\  /\  /\  /\
/__\/__\/__\/__\

以此类推，输出规模为 nnn 的图形。

思路：已知初始时 n=1n=1n=1 的图形：

a[1][2] = a[2][1] = '/';
a[1][3] = a[2][4] = '\\'; //转义
a[2][2] = a[2][3] = '_';

容易发现，规模为 n+1n+1n+1 的图形可以由规模为 nnn 的图形得到：

经过这一番扩展后，图形的宽、高均扩大为原来的 222 倍。

for (int i = 1; i <= h; i ++)
    for (int j = 1; j <= w; j ++)
        a[i + h][j] = a[i + h][j + w] = a[i][j];  //向下重复2次

for (int i = 1; i <= h; i ++)
    for (int j = w; j >= 1; j --)
    {
        a[i][j + w / 2] = a[i][j];  //向右平移半宽
        a[i][j] = ' '; //平移后，原位置就置空
    }
        
w *= 2, h *= 2; //更新宽、高

由规模为 111 的图形，只需重复以上操作 kkk 次，就能得到规模为 1+k1+k1+k 的分形图。

#include <iostream>
#include <cstring>
using namespace std;

char a[3000][3000]; 

int main()
{
    int n; 
    cin >> n;

    memset(a, ' ', sizeof(a)); //初始时全部置为空格
    a[1][2] = a[2][1] = '/';
    a[1][3] = a[2][4] = '\\'; //转义
    a[2][2] = a[2][3] = '_';

    int h = 2, w = 4; //初始时的高、宽
    for (int t = 1; t < n; t ++)  //重复进行n-1次该操作
    {
        for (int i = 1; i <= h; i ++)
            for (int j = 1; j <= w; j ++)
                a[i + h][j] = a[i + h][j + w] = a[i][j];

        for (int i = 1; i <= h; i ++)
            for (int j = w; j >= 1; j --)
            {
                a[i][j + w / 2] = a[i][j];
                a[i][j] = ' ';
            }
        
        w *= 2, h *= 2;
    }

    for (int i = 1; i <= h; i ++)
    {
        for (int j = 1; j <= w; j ++) cout << a[i][j];
        puts("");
    }
    return 0;
}
