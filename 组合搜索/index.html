组合搜索

· 2022-09-02 · # 枚举 # 递归

原题链接：选数

题目大意：从 nnn 个数中任选 kkk 个相加，求所得的和中有多少个质数。

思路：本题就是在考察如何在 nnn 个数中枚举 kkk 元组，显然方案数有：CnkC_n^kCnk​ 种。

搜索时关键是要做到不重不漏。关键就是要做到“不减遍历”，即本次搜索到的 kkk 个下标一定比 上次搜索到的 kkk 个下标要大。直观上看就是 kkk 个指针都在逐渐右移。

参考：用递归进行枚举

#include <iostream>
using namespace std;

const int N = 22;
int a[N], n, k, res;
bool st[N];

bool isPrime(int x)
{
    if (x < 2) return false;
    for (int i = 2; i <= x / i; i ++)
        if (x % i == 0) return false;
    return true;
}

// u记录当前已经枚举的元素个数，sum为当前的元素和，begin记录枚举的起始下标
void dfs(int u, int sum, int begin) 
{
    if (u == k) 
    {
        if (isPrime(sum)) res ++;
        return;
    }

    for (int i = begin; i < n; i ++)
        if (st[i] == false)
        {
            st[i] = true;
            dfs(u + 1, sum + a[i], i + 1);
            st[i] = false;
        }
}

int main()
{
    cin >> n >> k;
    for (int i = 0; i < n; i ++) cin >> a[i];

    dfs(0, 0, 0);

    cout << res;
    return 0;
}
