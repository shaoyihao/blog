旋转矩阵

· 2022-09-09 · # 模拟

原题链接：旋转矩阵

题目大意：给定一个 n×nn\times nn×n 的矩阵，对于 x,y,r,zx,y,r,zx,y,r,z，把以第 xxx 行第 yyy 列的元素为中心的 2r+12r+12r+1 阶矩阵按照某种时针方向旋转（z=0z=0z=0 表示顺时针，z=1z=1z=1 表示逆时针）。

思路：

在模拟时，如果是在原矩阵上直接进行覆盖，则一定要注意覆盖的顺序：因为2条邻边之间共享了一个元素（拐角元素）。

正确的覆盖顺序也是一种非常符合直觉的方式。（脑中动态模拟一下）

事实上只要前333边的覆盖满足上述顺序即可，最后一条边可以不必这样，因为最后一条边已经是所剩唯一还没被处理的边，覆盖它不会影响到其他边。

#include <iostream>
using namespace std;

const int N = 510;
int g[N][N];

void f(int x, int y, int r, int z)
{
    int tmp[2 * r + 1];
    for (int i = 0; i < 2 * r + 1; i ++)
        tmp[i] = g[x - r][y - r + i];  //暂存一条边，因为该边接下来会被覆盖

    if (z == 1) //逆时针
    {
        for (int i = 0; i < 2 * r + 1; i ++) //上
            g[x - r][y - r + i] = g[x - r + i][y + r];
        for (int i = 0; i < 2 * r + 1; i ++) //右
            g[x - r + i][y + r] = g[x + r][y + r - i];
        for (int i = 0; i < 2 * r + 1; i ++) //下
            g[x + r][y + r - i] = g[x + r - i][y - r];
        for (int i = 0; i < 2 * r + 1; i ++) //左
            g[x + r - i][y - r] = tmp[i];
    }
    else //顺时针
    {
        for (int i = 0; i < 2 * r + 1; i ++) //上
            g[x - r][y + r - i] = g[x - r + i][y - r];
        for (int i = 0; i < 2 * r + 1; i ++) //左
            g[x - r + i][y - r] = g[x + r][y - r + i];
        for (int i = 0; i < 2 * r + 1; i ++) //下
            g[x + r][y - r + i] = g[x + r - i][y + r];
        for (int i = 0; i < 2 * r + 1; i ++) //右
            g[x - r + i][y + r] = tmp[i]; //最后一条边随意些也无妨
    }
}

int main()
{
    int n, m;
    cin >> n >> m;
    for (int i = 1, idx = 1; i <= n; i ++)
        for (int j = 1; j <= n; j ++) 
            g[i][j] = idx ++;
    
    while (m --)
    {
        int x, y, r, z;
        cin >> x >> y >> r >> z;
        while (r) //一个矩阵可以视作是多个“环”
        {
            f(x, y, r, z);
            r --;
        }
    }

    for (int i = 1; i <= n; i ++)
    {
        for (int j = 1; j <= n; j ++) cout << g[i][j] << ' ';
        puts("");
    }
    return 0;
}

当然，如果每次都将矩阵拷贝一遍，再进行复制，那么就不必考虑可能覆盖的问题了。使用了额外空间，但是逻辑上更加清晰了。

#include <iostream>
#include <cstring>
using namespace std;

const int N = 510;
int g[N][N], g2[N][N];

void f(int x, int y, int r, int z)
{
    if (z == 1) //逆时针
    {
        for (int i = 0; i < 2 * r + 1; i ++)  //上
            g[x - r][y - r + i] = g2[x - r + i][y + r];
        for (int i = 0; i < 2 * r + 1; i ++)  //右
            g[x - r + i][y + r] = g2[x + r][y + r - i];
        for (int i = 0; i < 2 * r + 1; i ++)  //下
            g[x + r][y - r + i] = g2[x - r + i][y - r];
        for (int i = 0; i < 2 * r + 1; i ++)  //左
            g[x - r + i][y - r] = g2[x  - r][y + r - i];
    }
    else //顺时针
    {
        for (int i = 0; i < 2 * r + 1; i ++)  //上
            g[x - r][y - r + i] = g2[x + r - i][y - r];
        for (int i = 0; i < 2 * r + 1; i ++)  //左
            g[x - r + i][y - r] = g2[x + r][y - r + i];
        for (int i = 0; i < 2 * r + 1; i ++)  //下
            g[x + r][y - r + i] = g2[x + r - i][y + r];
        for (int i = 0; i < 2 * r + 1; i ++)  //右  
            g[x - r + i][y + r] = g2[x - r][y - r + i];
    }
}

int main()
{
    int n, m;
    cin >> n >> m;
    for (int i = 1, idx = 1; i <= n; i ++)
        for (int j = 1; j <= n; j ++) 
            g[i][j] = idx ++;
    
    while (m --)
    {
        memcpy(g2, g, sizeof(g)); //拷贝当前矩阵的状态
        
        int x, y, r, z;
        cin >> x >> y >> r >> z;
        while (r) //旋转该矩阵：一个矩阵可以视作是多个“环”
        {
            f(x, y, r, z); // 旋转一个环
            r --;
        }
    }

    for (int i = 1; i <= n; i ++)
    {
        for (int j = 1; j <= n; j ++) cout << g[i][j] << ' ';
        puts("");
    }
    return 0;
}
